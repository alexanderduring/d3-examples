Data Join with Spans

        
function updateParagraphs()
{
    var dataValues = d3.select("#data").property("value");
    var data = JSON.parse(dataValues);

    var updatedElements = d3.select("#elements").selectAll("span").data(data);

    updatedElements.enter()
        .append("span")
        .text(function (d) { return d; });

    updatedElements.exit()
        .remove();

    updatedElements.text(function (d) { return d; });

}
